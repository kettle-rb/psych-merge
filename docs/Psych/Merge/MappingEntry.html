def content
  return "" unless start_line && end_line

  (start_line..end_line).map { |ln| @lines[ln - 1] }.compact.join("\n")
end

def start_line
  # Include leading comments in start line
  leading = @comment_tracker.leading_comments_before(@key.start_line || 1)
  if leading.any?
    leading.first[:line]
  else
    @key.start_line
  end
end
